Parametros e variáveis

function soma() {
    let soma = 0
    for (i in arguments) { //arguments é um array, usado para adicionar parametros à funcao
        soma += arguments[i] //vai somar todos os parametros da função
    }
    return soma
}

console.log(soma())
// 0
console.log(soma(1))
// 1
console.log(soma(1.1, 2.2, 3.3))
// 6.6
console.log(soma(1.1, 2.2, "Teste"))
// 3.30000000003teste
console.log(soma('a', 'b', 'c'))
// 0abc
