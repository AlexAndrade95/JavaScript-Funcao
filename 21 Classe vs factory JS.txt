Classe vs factory
classe:
class Pessoa {
    constructor(nome) {
        this.nome = nome
    }

    falar() {
        console.log(`Meu nome é ${this.nome}`)
    }
}

const p1 = new Pessoa('João')
p1.falar()
// Meu nome é João // da problema no browser pq usa this

funcão factory:
const criarPessoa = nome => {
    return {
        falar: () => console.log(`Meu nome é ${nome}`) 
    }
}

const p2 = criarPessoa('João')
p2.falar()
// Meu nome é João // não da problema no browser pq não usa this

factory:

function Pessoa (nome)  {
	this(nome) = nome
        this.nome = nome
    }

    this.falar()= function {
        console.log(`Meu nome é ${this.nome}`)
    }
}

const p1 = new Pessoa('João')
p1.falar()
// Meu nome é João
